Partie 1 — Notions de Base & Classes Simples

À quoi ça sert ? Représenter les objets "données" du sujet (Chambre, Gare, etc.). C'est souvent la première question de l'annale.

Modèle : Attributs privés, Constructeurs, Getters/Setters.

src/Base.java
1/**
2 * RÔLE DU FICHIER : Représenter une classe utilitaire simple.
3 * MODÈLE D'ANNALE : Correspond aux classes comme "Chambre", "Gare", "Etape".
4 *
5 * NOTION : Classe simple, attributs privés, constructeurs, getters/setters, toString.
6 */
7public class Base {
8
9 // Attributs : Ce que l'objet "connaît" d'elle-même.
10 // private = encapsulation (protection des données)
11 private String coordGPS;
12 private double volumeHangar;
13
14 /**
15 * CONSTRUCTEUR VIDE (par défaut)
16 * Règle d'examen : Toujours utile d'en avoir un si on ne veut pas forcer les paramètres.
17 */
18 public Base() {
19 }
20
21 /**
22 * CONSTRUCTEUR PARAMÉTRÉ
23 * Sert à initialiser l'objet dès sa création avec 'new'.
24 * @param coordGPS La position géographique.
25 * @param volumeHangar La taille du hangar.
26 */
27 public Base(String coordGPS, double volumeHangar) {
28 this.coordGPS = coordGPS;
29 this.volumeHangar = volumeHangar;
30 }
31
32 // GETTERS : Pour "lire" la valeur d'un attribut privé.
33 public String getCoordGPS() { return coordGPS; }
34 public double getVolumeHangar() { return volumeHangar; }
35
36 // SETTERS : Pour "modifier" la valeur d'un attribut privé.
37 public void setCoordGPS(String coordGPS) { this.coordGPS = coordGPS; }
38 public void setVolumeHangar(double volumeHangar) { this.volumeHangar = volumeHangar; }
39
40 /**
41 * TOSTRING : Comment l'objet s'affiche dans la console (ex: System.out.println).
42 */
43 @Override
44 public String toString() {
45 return "Base{" + "coordGPS='" + coordGPS + '\'' + ", volumeHangar=" + volumeHangar + '}';
46 }
47}
